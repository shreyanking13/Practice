#include <bits/stdc++.h>
using namespace std;



int main() 
{
    char input = 'A';
    switch( input ){
        case 'A': {
            cout << "excellent";
            break;
        }   
        case 'B': {
            cout << "very good";
            break;
        } 
        case 'C': {
            cout << "good";
            break;
        } 
        case 'D': {
            cout << "not bad";
            break;
        } 
    } 
    
    return 0;
}
