using namespace std;

// return type is void 
void print_array(  vector<int> arr ) {

    for( int i = 0 ; i <= arr.size() - 1 ; i++){
        cout << arr[i] << " " ;

    } 
    cout <<   endl;


}

// return type is int 
int sum_of_all_element( vector<int> arr ) {
    int sum = 0;
    for( int i = 0 ; i <= arr.size() - 1 ; i++){
        // sum = sum + arr[i]
       sum += arr[i];

    }     
    return sum;

} 

int main() 
{   
    int size;
    cin >> size;
    // dynamic array where size can be changed using push back and pop back
    vector<int> arr(size);

    for( int i = 0 ; i <= size - 1 ; i++){
        cin >> arr[i]; 

    }
    // print array function is called and arr variable is passedâ€Ž
    print_array( arr ) ;
    cout << " sum_of_all_element: " << sum_of_all_element( arr ) << endl;

    arr.push_back(10);
    arr.push_back(11);

    print_array( arr ) ;
    cout << " sum_of_all_element: " << sum_of_all_element( arr ) << endl;


    arr.pop_back();
    arr.pop_back();
    arr.pop_back();

    print_array( arr ) ;
    cout << " sum_of_all_element: " << sum_of_all_element( arr ) << endl;


    return 0;

}
